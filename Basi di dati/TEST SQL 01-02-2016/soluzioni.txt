2)
select distinct t1.nome,t1.cognome,t2.nome,t2.cognome
from traduttore as t1, traduttore as t2
where t1.madrelingua != t2.madrelingua and t1.nome > t2.nome

3)
select traduttore.codice, count(*)
from traduttore join traduzione on traduttore.codice = traduzione.codtraduttore
where traduzione.linguaorigine = 'inglese' and traduzione.linguadestinazione = traduttore.madrelingua
group by traduttore.codice

from traduzione join traduttore on traduzione.codtraduttore = traduttore.codice
group by traduzione.brano,traduzione.codtraduttore
order by traduzione.codtraduttore)
